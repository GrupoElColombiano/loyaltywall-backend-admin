import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Plan } from './entity/plan.entity';
import { PlansProductCategory } from 'src/common/entity/plans-products-categories.entity';

@Injectable()
export class PlansService {
  constructor(
    @InjectRepository(Plan) private readonly planRepository: Repository<Plan>,
    @InjectRepository(PlansProductCategory)
    private readonly plansProductCategoryRepository: Repository<PlansProductCategory>,
  ) {}

  /**
   * Create a new plan.
   * @param plan - Plan data to create.
   * @returns {Promise<any>} A promise that resolves to the created plan.
   */
  async create(plan: any): Promise<any> {
    try {
      const planCreated = await this.planRepository.findOne({
        where: { name: plan.name },
      });

      if (planCreated) {
        throw new NotFoundException(
          `Plan with name ${plan.name} already exists`,
        );
      }

      const planEdited: any = {
        userType: plan.userType,
        name: plan.name,
        description: plan.description,
        isActive: plan.isActive,
        idVersionPlan: plan.idVersionPlan,
        time: plan.time,
        typeDuration: plan.typeDuration,
        duration: plan.duration,
        rate: plan.rate,
        renewalRate: plan.renewalRate,
        isSpecialRateActive: plan.isSpecialRateActive,
        specialRate: plan.specialRate,
        renewalSpecialRate: plan.renewalSpecialRate,
        startDate: plan.startDate,
        endDate: plan.endDate,
      };

      const newPlan = await this.planRepository.save(planEdited);
      console.log('newPlan', newPlan);

      plan.categories.map(async (item: any) => {
        console.log('item', item);
        const newItems: any = {
          plan: newPlan.idPlan,
          sites: item.sites,
          idProduct: item.idProduct,
          idCategory: item.idCategory,
          duration: item.duration,
          typeDuration: item.typeDuration,
          frequency: item.frequency,
          unlimited: item.unlimited,
          amount: item.amount,
        };
        console.log('newItems', newItems);
        const franck = await this.plansProductCategoryRepository.save(newItems);
      });

      return {
        message: 'Plan created successfully',
        newPlan,
      };
    } catch (error) {
      throw new NotFoundException(error.message);
    }
  }

  /**
   * Update a plan by ID.
   * @param id - Plan ID.
   * @param updatedPlan - Updated plan data.
   * @returns {Promise<any>} A promise that resolves to the updated plan.
   */
  async updatePlan(id: number, updatedPlan: any): Promise<any> {
    try {
      const planToUpdate = await this.planRepository.findOne({
        where: { idPlan: id },
      });

      if (!planToUpdate) {
        throw new NotFoundException(`Plan with ID ${id} not found`);
      }

      const {
        name,
        userType,
        description,
        isActive,
        idVersionPlan,
        categories,
      } = updatedPlan;

      // Actualizar los campos del plan con los valores proporcionados
      planToUpdate.name = name;
      planToUpdate.userType = userType;
      planToUpdate.description = description;
      planToUpdate.isActive = isActive;
      planToUpdate.idVersionPlan = idVersionPlan;

      // Guardar los cambios en el plan
      const updatedPlanResult = await this.planRepository.save(planToUpdate);

      // Actualizar las categorías asociadas con el plan
      for (const category of categories) {
        const {
          sites,
          idProduct,
          idCategory,
          duration,
          typeDuration,
          frequency,
          unlimited,
          amount,
        } = category;

        const newCategory = {
          plan: updatedPlanResult, // Asociamos la categoría con el plan actualizado
          sites,
          idProduct,
          idCategory,
          duration,
          typeDuration,
          frequency,
          unlimited,
          amount,
        };

        await this.plansProductCategoryRepository.save(newCategory);
      }

      return {
        message: 'Plan updated successfully',
        updatedPlan: updatedPlanResult,
      };
    } catch (error) {
      throw new NotFoundException(error.message);
    }
  }

  /**
   * Find all plans.
   * @returns {Promise<any>} A promise that resolves to a list of plans.
   */
  async findAllPlans(): Promise<any> {
    try {
      const plans = await this.planRepository.find();
      const totalPlans = await this.planRepository.count();
      return {
        data: plans,
        totalPlans,
      };
    } catch (error) {}
  }

  /**
   * Find all plans with pagination support.
   * @param page - Page number.
   * @param limit - Number of items per page.
   * @returns {Promise<any>} A promise that resolves to a paginated list of plans.
   */
  async findAll(page: number, limit: number): Promise<Plan[] | any> {
    try {
      const skip = (page - 1) * limit;
      const [plans, totalPlans] = await this.planRepository.findAndCount({
        skip,
        take: limit,
      });
      const totalPages = Math.ceil(totalPlans / limit);

      if (page > totalPages) {
        throw new NotFoundException('Number of page is not valid');
      }

      return {
        data: plans,
        totalPlans,
      };
    } catch (error) {
      throw new NotFoundException(error.message);
    }
  }

  /**
   * Find a plan by its ID.
   * @param id - Plan ID.
   * @returns {Promise<Plan>} A promise that resolves to the found plan.
   */
  async findOne(id: number): Promise<Plan> {
    const plan = await this.planRepository.findOne({
      where: { idPlan: id },
    });

    if (!plan) {
      throw new NotFoundException(`Plan with ID ${id} not found`);
    }

    return plan;
  }

  /**
   * Update the isActive status of a plan.
   * @param id - Plan ID.
   * @param isActive - New isActive status.
   * @returns {Promise<Plan | any>} A promise that resolves to the updated plan.
   */
  async updateIsActive(id: number, isActive: any): Promise<Plan | any> {
    try {
      const planToUpdate = await this.planRepository.findOne({
        where: { idPlan: id },
      });

      if (!planToUpdate) {
        throw new NotFoundException(`Plan with ID ${id} not found`);
      }

      // Actualizar el campo isActive con el nuevo valor
      planToUpdate.isActive = isActive;

      // Guardar los cambios en la base de datos
      const updatedPlan = await this.planRepository.save(planToUpdate);

      return {
        message: 'Plan updated successfully',
        updatedPlan,
      };
    } catch (error) {
      throw new NotFoundException(error.message);
    }
  }

  /**
   * Find plans based on filtering criteria.
   * @param query - Filtering criteria.
   * @returns {Promise<Plan[]>} A promise that resolves to a list of filtered plans.
   */
  async findByFilter(query: any): Promise<Plan[]> {
    const { name, rate, duration } = query;

    try {
      const queryBuilder = this.planRepository.createQueryBuilder('plan');

      if (name) {
        queryBuilder.where('LOWER(plan.name) LIKE :name', {
          name: `%${name.toLowerCase()}%`,
        });
      }

      if (rate) {
        queryBuilder.where('LOWER(plan.rate) LIKE :rate', {
          rate: `%${rate.toLowerCase()}%`,
        });
      }

      if (duration) {
        queryBuilder.where('LOWER(plan.duration) LIKE :duration', {
          duration: `%${duration.toLowerCase()}%`,
        });
      }

      const plans = await queryBuilder.getMany();
      console.log('plans', plans);
      return plans;
    } catch (error) {
      console.log('error', error);
      throw new NotFoundException(error.message);
    }
  }

  //Unión entre findByFilter y findAll
  /**
   * Find plans based on filtering criteria with pagination support.
   * @param query - Filtering and pagination criteria.
   * @returns {Promise<any>} A promise that resolves to a paginated list of filtered plans.
   */
  async findByFilterAndPagination(query: any): Promise<any> {
    const { name, rate, duration, page, limit } = query;

    try {
      const skip = (page - 1) * limit;
      const queryBuilder = this.planRepository
        .createQueryBuilder('plan')
        .skip(skip)
        .take(limit);

      if (name) {
        queryBuilder.where('LOWER(plan.name) LIKE :name', {
          name: `%${name.toLowerCase()}%`,
        });
      }

      if (rate) {
        queryBuilder.where('LOWER(plan.rate) LIKE :rate', {
          rate: `%${rate.toLowerCase()}%`,
        });
      }

      if (duration) {
        queryBuilder.where('LOWER(plan.duration) LIKE :duration', {
          duration: `%${duration.toLowerCase()}%`,
        });
      }

      const [plans, totalPlans] = await queryBuilder.getManyAndCount();
      const totalPages = Math.ceil(totalPlans / limit);

      if (page > totalPages) {
        throw new NotFoundException('Number of page is not valid');
      }

      return {
        data: plans,
        totalPlans,
      };
    } catch (error) {
      throw new NotFoundException(error.message);
    }
  }
}

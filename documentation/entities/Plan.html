// import { PlansProducts } from '../../common/entity/plans-product.entity';
import { Product } from '../../product/entity/product.entity';
import { PlansProductCategory } from '../../common/entity/plans-products-categories.entity';
import {
  Entity,
  Column,
  PrimaryGeneratedColumn,
  CreateDateColumn,
  UpdateDateColumn,
  OneToMany,
} from 'typeorm';

@Entity({ name: 'plans' })
export class Plan {
  @PrimaryGeneratedColumn({ name: 'idPlan' })
  idPlan?: number;

  @Column({ name: 'idVersionPlan', default: 1 })
  idVersionPlan: number = 1;

  @Column()
  description: string;

  @Column()
  name: string;

  @Column()
  userType: string;

  @Column()
  isActive: boolean;

  @Column({ nullable: true })
  time: string;

  @Column({ name: 'typeDuration', nullable: true })
  typeDuration: string;

  @Column({ name: 'duration', nullable: true })
  duration: number;

  @Column({ nullable: true })
  rate: number;

  @Column({ name: 'renewal_rate', nullable: true })
  renewalRate: number;

  @Column({ name: 'special_rate_enabled', default: false })
  isSpecialRateActive: boolean;

  @Column({ name: 'special_rate', nullable: true })
  specialRate: number;

  @Column({ name: 'renewal_special_rate', nullable: true })
  renewalSpecialRate: number;

  @Column({ name: 'start_date', type: 'date', nullable: true })
  startDate: Date;

  @Column({ name: 'end_date', type: 'date', nullable: true })
  endDate: Date;

  @CreateDateColumn({ name: 'created_at', type: 'timestamp' })
  createdAt: Date;

  @UpdateDateColumn({ name: 'updated_at', type: 'timestamp' })
  updatedAt: Date;

  @OneToMany(() => PlansProductCategory, plansProductsCategory => plansProductsCategory.plan)
  plansProductsCategory: PlansProductCategory[];
}
